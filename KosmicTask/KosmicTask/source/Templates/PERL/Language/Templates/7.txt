use warnings;

# capture screen image to file
# files created in the current directory will be deleted when the task ends
$FILE="capture.png";

# define command
$COMMAND="screencapture";

# log command
print STDERR "Running $COMMAND command...\n";

# execute command
$STATUS=system("$COMMAND -t png $FILE");

# check for errors
if ($STATUS != 0) {
	print "--- {kosmicError: 'Error running $COMMAND'}";
	exit 1;
}

# return a YAML inline format dictionary with filename
print "--- {kosmicFile: $FILE, kosmicInfo: file returned}";
