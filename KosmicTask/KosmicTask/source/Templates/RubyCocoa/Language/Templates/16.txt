require 'osx/cocoa'
include OSX
OSX.require_framework 'ScriptingBridge'

def kosmictask()

	# get application reference
	app = OSX::SBApplication.applicationWithBundleIdentifier_("com.apple.finder")

	# query application
	return app.name
end
