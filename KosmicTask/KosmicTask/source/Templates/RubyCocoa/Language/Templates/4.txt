require 'osx/cocoa'

OSX.require_framework "AddressBook"

def kosmictask()

	person = OSX::ABAddressBook.sharedAddressBook().me()
	firstName = person.valueForProperty_(OSX::KABFirstNameProperty)

	return firstName
end
