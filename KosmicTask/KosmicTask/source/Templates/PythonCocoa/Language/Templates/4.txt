import AppKit

# import our system framework by name
import AddressBook

def kosmictask(a , b):
	
	AppKit.NSLog("task parameters are %@ and %@", a, b)
	
	person = AddressBook.ABAddressBook.sharedAddressBook().me()
	firstName = person.valueForProperty_(AddressBook.kABFirstNameProperty)
	
	# return object
	return firstName
