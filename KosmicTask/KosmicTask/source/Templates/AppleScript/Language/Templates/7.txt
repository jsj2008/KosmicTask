on KosmicTask()
	
	try
		
		-- compute task result
		set myResult to {key1:"item 1", key2:"item 2"}
		
		-- set style string
		set myStyle to "color:green;font-weight:bold;font-style:italic;text-decoration:underline;font-size:18pt;"

	end try

end KosmicTask
