on KosmicTask()
	
	try
		
		-- compute task result
		set myResult to {"item 1", "item 2"}
		
		-- return result record
		return {kosmicData:myResult}
		
	on error errorMessage number errorNumber

		return {kosmicError:errorMessage}

	end try

end KosmicTask
