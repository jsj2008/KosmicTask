# capture screen image to file
# files created in the current directory will be deleted when the task ends
set files = ""
set i = 1

while ( $i <= 5)

	if ( $i > 1) then
		set files = "$files , "
	endif
	
	# filename
	set file = "capture-$i.png"

	# capture screen
	screencapture -t png $file
	
	# add to files list
	set files = "$files $file"
	
	sleep 1
	
	@ i++
end

# return a YAML inline format dictionary
echo "---"
echo "{kosmicFile: [$files] , kosmicInfo: file returned}"
